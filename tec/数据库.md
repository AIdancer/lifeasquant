### 分钟线数据表
```sql
drop table if exists fts_minute;
create table fts_minute (
	dataid int primary key not null auto_increment,
	timestamp decimal(13,3),
	instrument varchar(16),
	open decimal(10,2),
	high decimal(10,2),
	low decimal(10,2),
	close decimal(10,2),
	volume int
) default charset=utf8;
```

### tick数据表
```sql
drop table if exists fts_tick;
create table fts_tick (
	instrument varchar(16) primary key not null,
	timestamp decimal(13,3),
	open decimal(10,2),
	high decimal(10,2),
	low decimal(10,2),
	last decimal(10,2),
	volume int,
	avg decimal(10,2),
	ask1 decimal(10,2),
	volask1 int,
	bid1 decimal(10,2),
	volbid1 int
) default charset=utf8;
```
